# R11.1 — Multi-Project Fleet View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a meta-dashboard that shows all local projects at once — health, active agents, burn rate, blockers, and cost — so the user can answer "what's the status of everything I'm working on?" without switching project context.

**Architecture:** SQLite `project_health_snapshot` materialized table (refreshed on demand) feeds a `/api/fleet` REST endpoint; `discovered_projects` table enables passive git-directory scanning. A new `FleetView` React component becomes the default view when 2+ projects exist, rendering a dense project-card grid with sparklines and health chips. Two new MCP tools let agents query fleet state.

**Tech Stack:** TypeScript, Node.js, Express, better-sqlite3, Zod, React, Vite, vitest

**Design spec:** `docs/plans/R11/R11.1-multi-project-fleet-view.md`

  ```typescript
  import Database from "better-sqlite3";
  import { readdirSync, existsSync } from "fs";
  import { join } from "path";
  import { genId, now } from "./helpers.js";
  import type {
    FleetProjectSummary,
    FleetOverview,
    DiscoveredProject,
  } from "../types.js";

  const ACTIVE_AGENT_WINDOW_MS = 5 * 60 * 1000;

  function computeHealth(
    open_blockers: number,
    overdue_tasks: number
  ): "on_track" | "at_risk" | "behind" {
    if (open_blockers > 0 || overdue_tasks > 2) return "behind";
    if (overdue_tasks > 0) return "at_risk";
    return "on_track";
  }

  export function refreshFleetSnapshots(db: Database.Database): void {
    const projects = db
      .prepare("SELECT id FROM projects")
      .all() as { id: string }[];

    const activeCutoff = new Date(
      Date.now() - ACTIVE_AGENT_WINDOW_MS
    ).toISOString();
    const cutoff7d = new Date(
      Date.now() - 7 * 24 * 60 * 60 * 1000
    ).toISOString();
    const cutoff30d = new Date(
      Date.now() - 30 * 24 * 60 * 60 * 1000
    ).toISOString();
    const nowIso = new Date().toISOString();
    const snapshotAt = now();

    const upsert = db.prepare(`
      INSERT INTO project_health_snapshot
        (project_id, snapshot_at, active_agents, open_blockers, overdue_tasks,
         in_progress_tasks, cost_7d, cost_30d, milestone_progress_pct)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
      ON CONFLICT(project_id) DO UPDATE SET
        snapshot_at = excluded.snapshot_at,
        active_agents = excluded.active_agents,
        open_blockers = excluded.open_blockers,
        overdue_tasks = excluded.overdue_tasks,
        in_progress_tasks = excluded.in_progress_tasks,
        cost_7d = excluded.cost_7d,
        cost_30d = excluded.cost_30d,
        milestone_progress_pct = excluded.milestone_progress_pct
    `);

    const stmtActiveAgents = db.prepare(`
      SELECT COUNT(DISTINCT a.id) as count
      FROM agents a
      INNER JOIN tasks t ON t.assigned_agent_id = a.id
      WHERE t.project_id = ? AND a.last_seen_at >= ?
    `);
    const stmtBlockers = db.prepare(`
      SELECT COUNT(*) as count FROM blockers
      WHERE resolved_at IS NULL
        AND task_id IN (SELECT id FROM tasks WHERE project_id = ?)
    `);
    const stmtOverdue = db.prepare(`
      SELECT COUNT(*) as count FROM tasks
      WHERE project_id = ? AND status != 'done'
        AND due_date IS NOT NULL AND due_date < ?
    `);
    const stmtInProgress = db.prepare(`
      SELECT COUNT(*) as count FROM tasks
      WHERE project_id = ? AND status = 'in_progress'
    `);
    const stmtCost7d = db.prepare(`
      SELECT COALESCE(SUM(cost_usd), 0) as total FROM cost_entries
      WHERE project_id = ? AND created_at >= ?
    `);
    const stmtCost30d = db.prepare(`
      SELECT COALESCE(SUM(cost_usd), 0) as total FROM cost_entries
      WHERE project_id = ? AND created_at >= ?
    `);
    const stmtMilestoneProgress = db.prepare(`
      SELECT COALESCE(AVG(
        CASE WHEN m.status = 'achieved' THEN 100.0
             ELSE CAST(t_done.done_count AS REAL) / NULLIF(t_all.all_count, 0) * 100.0
        END
      ), 0) as avg_pct
      FROM milestones m
      LEFT JOIN (
        SELECT milestone_id, COUNT(*) as done_count
        FROM tasks WHERE status = 'done' GROUP BY milestone_id
      ) t_done ON t_done.milestone_id = m.id
      LEFT JOIN (
        SELECT milestone_id, COUNT(*) as all_count
        FROM tasks GROUP BY milestone_id
      ) t_all ON t_all.milestone_id = m.id
      WHERE m.project_id = ? AND m.status != 'achieved'
    `);

    for (const { id: projectId } of projects) {
      const activeAgents = (
        stmtActiveAgents.get(projectId, activeCutoff) as { count: number }
      ).count;
      const openBlockers = (
        stmtBlockers.get(projectId) as { count: number }
      ).count;
      const overdueTasks = (
        stmtOverdue.get(projectId, nowIso) as { count: number }
      ).count;
      const inProgressTasks = (
        stmtInProgress.get(projectId) as { count: number }
      ).count;
      const cost7d = (stmtCost7d.get(projectId, cutoff7d) as { total: number })
        .total;
      const cost30d = (
        stmtCost30d.get(projectId, cutoff30d) as { total: number }
      ).total;
      const milestoneProgressPct = (
        stmtMilestoneProgress.get(projectId) as { avg_pct: number }
      ).avg_pct;

      upsert.run(
        projectId,
        snapshotAt,
        activeAgents,
        openBlockers,
        overdueTasks,
        inProgressTasks,
        cost7d,
        cost30d,
        milestoneProgressPct
      );
    }
  }

  export function getFleetOverview(db: Database.Database): FleetOverview {
    const rows = db
      .prepare(
        `SELECT p.id as project_id, p.name as project_name,
                s.snapshot_at, s.active_agents, s.open_blockers, s.overdue_tasks,
                s.in_progress_tasks, s.cost_7d, s.cost_30d, s.milestone_progress_pct
         FROM projects p
         LEFT JOIN project_health_snapshot s ON s.project_id = p.id
         ORDER BY p.name ASC`
      )
      .all() as Array<{
      project_id: string;
      project_name: string;
      snapshot_at: string | null;
      active_agents: number | null;
      open_blockers: number | null;
      overdue_tasks: number | null;
      in_progress_tasks: number | null;
      cost_7d: number | null;
      cost_30d: number | null;
      milestone_progress_pct: number | null;
    }>;

    const stmtActivity = db.prepare(`
      SELECT al.message as event_type, al.timestamp as created_at, a.name as agent_name
      FROM activity_log al
      JOIN tasks t ON t.id = al.task_id
      LEFT JOIN agents a ON a.id = al.agent_id
      WHERE t.project_id = ?
      ORDER BY al.timestamp DESC LIMIT 5
    `);

    const projects: FleetProjectSummary[] = rows.map((row) => {
      const activeAgents = row.active_agents ?? 0;
      const openBlockers = row.open_blockers ?? 0;
      const overdueTasks = row.overdue_tasks ?? 0;
      const inProgressTasks = row.in_progress_tasks ?? 0;
      const cost7d = row.cost_7d ?? 0;
      const cost30d = row.cost_30d ?? 0;
      const milestoneProgressPct = row.milestone_progress_pct ?? 0;

      const health: FleetProjectSummary["health"] =
        row.snapshot_at === null
          ? "unknown"
          : computeHealth(openBlockers, overdueTasks);

      const recentActivity = stmtActivity.all(row.project_id) as Array<{
        event_type: string;
        created_at: string;
        agent_name: string | null;
      }>;

      return {
        project_id: row.project_id,
        project_name: row.project_name,
        health,
        active_agents: activeAgents,
        open_blockers: openBlockers,
        overdue_tasks: overdueTasks,
        in_progress_tasks: inProgressTasks,
        cost_7d: cost7d,
        cost_30d: cost30d,
        milestone_progress_pct: milestoneProgressPct,
        snapshot_at: row.snapshot_at,
        recent_activity: recentActivity,
      };
    });

    return {
      projects,
      total_active_agents: projects.reduce((s, p) => s + p.active_agents, 0),
      total_open_blockers: projects.reduce((s, p) => s + p.open_blockers, 0),
      total_cost_7d: projects.reduce((s, p) => s + p.cost_7d, 0),
      refreshed_at: new Date().toISOString(),
    };
  }

  export function listDiscoveredProjects(
    db: Database.Database
  ): DiscoveredProject[] {
    return db
      .prepare(
        "SELECT * FROM discovered_projects ORDER BY detected_at DESC"
      )
      .all() as DiscoveredProject[];
  }

  export function linkDiscoveredProject(
    db: Database.Database,
    discoveryId: string,
    projectId: string | null
  ): DiscoveredProject | null {
    const existing = db
      .prepare("SELECT id FROM discovered_projects WHERE id = ?")
      .get(discoveryId);
    if (!existing) return null;
    db.prepare(
      "UPDATE discovered_projects SET linked_project_id = ? WHERE id = ?"
    ).run(projectId, discoveryId);
    return db
      .prepare("SELECT * FROM discovered_projects WHERE id = ?")
      .get(discoveryId) as DiscoveredProject;
  }

  export function ignoreDiscoveredProject(
    db: Database.Database,
    discoveryId: string
  ): DiscoveredProject | null {
    const existing = db
      .prepare("SELECT id FROM discovered_projects WHERE id = ?")
      .get(discoveryId);
    if (!existing) return null;
    db.prepare(
      "UPDATE discovered_projects SET ignored = 1 WHERE id = ?"
    ).run(discoveryId);
    return db
      .prepare("SELECT * FROM discovered_projects WHERE id = ?")
      .get(discoveryId) as DiscoveredProject;
  }

  export function discoverProjects(
    db: Database.Database,
    roots: string[]
  ): number {
    const insert = db.prepare(`
      INSERT OR IGNORE INTO discovered_projects (id, path, detected_at, linked_project_id, ignored)
      VALUES (?, ?, ?, NULL, 0)
    `);

    let discovered = 0;
    for (const root of roots) {
      if (!existsSync(root)) continue;
      try {
        const entries = readdirSync(root, { withFileTypes: true });
        for (const entry of entries) {
          if (!entry.isDirectory()) continue;
          const dirPath = join(root, entry.name);
          const hasGit = existsSync(join(dirPath, ".git"));
          const hasMcp = existsSync(join(dirPath, ".mcp.json"));
          const hasClaude = existsSync(join(dirPath, "CLAUDE.md"));
          if (hasGit && (hasMcp || hasClaude)) {
            const result = insert.run(genId(), dirPath, now());
            if (result.changes > 0) discovered++;
          }
        }
      } catch {
        // skip directories we cannot read
      }
    }
    return discovered;
  }
  ```

  ```typescript
  import type { DiscoveredProject } from "../types.js";

  function DiscoveryRow({
    discovery,
    onLink,
    onIgnore,
  }: {
    discovery: DiscoveredProject;
    onLink: () => void;
    onIgnore: () => void;
  }) {
    const shortPath =
      discovery.path.length > 60
        ? `…${discovery.path.slice(-57)}`
        : discovery.path;

    return (
      <div
        style={{
          display: "flex",
          alignItems: "center",
          gap: "12px",
          padding: "6px 0",
          borderBottom: "1px solid var(--border)",
        }}
      >
        <span
          style={{
            flex: 1,
            fontSize: "12px",
            color: "var(--text-primary)",
            fontFamily: "monospace",
            overflow: "hidden",
            textOverflow: "ellipsis",
            whiteSpace: "nowrap",
          }}
          title={discovery.path}
        >
          {shortPath}
        </span>
        <button
          onClick={onLink}
          style={{
            fontSize: "11px",
            padding: "3px 8px",
            background: "var(--accent-blue, #6366f1)",
            border: "none",
            borderRadius: "4px",
            color: "#fff",
            cursor: "pointer",
            flexShrink: 0,
          }}
        >
          Link
        </button>
        <button
          onClick={onIgnore}
          style={{
            fontSize: "11px",
            padding: "3px 8px",
            background: "var(--bg-tertiary)",
            border: "1px solid var(--border)",
            borderRadius: "4px",
            color: "var(--text-muted)",
            cursor: "pointer",
            flexShrink: 0,
          }}
        >
          Ignore
        </button>
      </div>
    );
  }

  export function DiscoveryBanner() {
    const { fleet, dispatch } = useFleetState();
    const api = useApi();

    const unlinked = fleet.discoveries.filter(
      (d) => d.linked_project_id === null && d.ignored === 0
    );

    const handleLink = useCallback(
      async (discovery: DiscoveredProject) => {
        // Auto-create a project using the directory basename, then link to it.
        const basename = discovery.path.replace(/\\/g, "/").split("/").pop() ?? discovery.path;
        const createRes = await fetch("/api/projects", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ name: basename, description: null }),
        });
        if (!createRes.ok) {
          console.error("Failed to create project for discovery");
          return;
        }
        const { id: projectId } = (await createRes.json()) as { id: string };
        await api.linkDiscovery(discovery.id, projectId);
        dispatch({
          type: "SET_FLEET_DISCOVERIES",
          payload: fleet.discoveries.filter((d) => d.id !== discovery.id),
        });
      },
      [api, dispatch, fleet.discoveries]
    );

    const handleIgnore = useCallback(
      async (discovery: DiscoveredProject) => {
        await api.ignoreDiscovery(discovery.id);
        dispatch({
          type: "SET_FLEET_DISCOVERIES",
          payload: fleet.discoveries.filter((d) => d.id !== discovery.id),
        });
      },
      [api, dispatch, fleet.discoveries]
    );

    if (unlinked.length === 0) return null;

    return (
      <div
        style={{
          background: "var(--bg-secondary)",
          border: "1px solid var(--accent-yellow, #f59e0b)",
          borderRadius: "8px",
          padding: "12px 16px",
          marginBottom: "16px",
        }}
      >
        <div
          style={{
            fontSize: "11px",
            fontWeight: 600,
            letterSpacing: "0.07em",
            textTransform: "uppercase",
            color: "var(--accent-yellow, #f59e0b)",
            marginBottom: "8px",
          }}
        >
          {unlinked.length} Discovered Project{unlinked.length !== 1 ? "s" : ""}
        </div>
        {unlinked.map((d) => (
          <DiscoveryRow
            key={d.id}
            discovery={d}
            onLink={() => handleLink(d)}
            onIgnore={() => handleIgnore(d)}
          />
        ))}
      </div>
    );
  }
  ```
